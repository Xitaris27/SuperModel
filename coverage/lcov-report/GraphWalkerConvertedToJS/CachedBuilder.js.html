var BuilderBase = require('./BuilderBase.js');
var Objects = require('./Objects.js');
 
 
const CachedBuilder = ({
  cache = null,
  builderBase = BuilderBase()
} = {}) => ({//public abstract class CachedBuilder<B, T> extends BuilderBase<B, T> {
  cache,//T - note:in one case it's a HashMap of runtimeVertices and Vertices, in another, a runtime vertex
  objects: Objects(),
  builderBase,
 
  invalidateCache() {//void
    this.cache = null;
  },
 
  createCache() {},//protected abstract T createCache();
 
  //@Override
  build() {//public T build() 
    if (this.objects.isNull(cache)) {
      this.cache = this.createCache();
    }                     
    return this.cache;
  },
 
  //@Override
  setId(id) {//public B setId(String id)
    this.invalidateCache();
    return this.builderBase.setId(id);
  },
 
  //@Override
  setName(name) {//public B setName(String name) {
    this.invalidateCache();
    return this.builderBase.setName(name);
  },
 
  //@Override
  addRequirement(requirement) {  //public B addRequirement(Requirement requirement) {
    this.invalidateCache();
    return this.builderBase.addRequirement(requirement);
  },
 
  //@Override
  setRequirements(requirements) {//  public B setRequirements(Set<Requirement> requirements) {
    this.invalidateCache();
    return this.builderBase.setRequirements(requirements);
  },
 
  //@Override
  setProperties(properties) {//  public B setProperties(Map<String, Object> properties) {
    this.invalidateCache();

  },
 
  //@Override
  setProperty(key, value) {//  public B setProperty(String key, Object value) {
    this.invalidateCache();
    return this.builderBase.setProperty(key, value);
  }
});
